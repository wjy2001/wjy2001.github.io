golang执行系统命令

调用系统命令使用的是os/exec，里面的exec.Command函数。

golang下的os/exec包执行外部命令，它将os.StartProcess进行包装使得它更容易映射到stdin和stdout。这点和python下的command、os.system等功能是一样的。这里列两个具体操作的代码。其可以调用类LINUX系统下的shell命令，也可以在windows下调用cmd下的命令。具体方法分为直接调用和使用程序调用。

linux

分为两种,以执行ls -ltr为演示

所在目录

1.使用bash执行命令

package main

import (
	"bytes"
	"fmt"
	"log"
	"os/exec"
)

const ShellToUse = "bash"

func Shellout(command string) (error, string, string) {
	var stdout bytes.Buffer
	var stderr bytes.Buffer
	cmd := exec.Command(ShellToUse, "-c", command)
	cmd.Stdout = &stdout//绑定标准输出，并且这里的输出不需要进行转码，中文能正常显示。
	cmd.Stderr = &stderr//绑定标准错误
	err := cmd.Run()
	return err, stdout.String(), stderr.String()
}
func main() {
	err, out, errout := Shellout("ls -ltr")
	if err != nil {
		log.Printf("error: %v\n", err)
	}
	fmt.Println("--- stdout ---")
	fmt.Println(out)
	fmt.Println("--- stderr ---")
	fmt.Println(errout)
}

使用函数转码

package main

import (
	"bytes"
	"fmt"
	"golang.org/x/text/encoding/simplifiedchinese"
	"log"
	"os/exec"
)

func Shellout(command string) (error, string, string) {
	var stdout bytes.Buffer
	var stderr bytes.Buffer
	cmd := exec.Command("cmd", "/c", command)
	cmd.Stdout = &stdout
	cmd.Stderr = &stderr
	err := cmd.Run()
	return err, stdout.String(), stderr.String()
}

//cmd程序需要使用这个转码成UTF-8，git不需要
func GBK2UTF8(str string) string {
	var decodeBytes, _ = simplifiedchinese.GB18030.NewDecoder().Bytes([]byte(str))
	return string(decodeBytes)
}

func main() {
	err, out, errout := Shellout("dir")
	if err != nil {
		log.Printf("error: %v\n", err)
	}
	fmt.Println("--- stdout ---")
	fmt.Println(GBK2UTF8(out))
	fmt.Println("--- stderr ---")
	fmt.Println(errout)
}

package main

import (
	"bytes"
	"fmt"
	"log"
	"os/exec"
)

	var stdout bytes.Buffer
	var stderr bytes.Buffer

	cmd.Stderr = &stderr
