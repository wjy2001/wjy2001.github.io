//将文档添加到结果之前要跳过的文档数。默认值为0。

所有

SetAllowDiskUse(b bool)
//如果为true，服务器可以在执行查找操作时将临时数据写入磁盘。此选项仅适用于MongoDB版本>=4.4
//如果指定了此选项，服务器版本>=3.2将报告错误。对于服务器版本<3.2，驱动程序将返回客户端错误。
//默认值是false。

SetAllowPartialResults(b bool)
//如果为true，则如果某些碎片已关闭而不是已关闭，则对碎片群集的操作可能会返回部分结果
//返回一个错误。默认值为false。

SetBatchSize(i int32)
//服务器返回的每个批中包含的最大文档数。

SetCollation(collation *collation)
//指定操作期间用于字符串比较的排序规则。此选项仅对MongoDB有效
//版本>=3.4。对于以前的服务器版本，如果使用此选项，驱动程序将返回错误。这个
//默认值为nil，这意味着将使用集合的默认排序规则。

SetComment(comment string)
//一个字符串，将包含在服务器日志、分析日志和currentOp查询中，以帮助跟踪操作。
//默认值为空字符串，这意味着日志中不会包含任何注释。

SetCursorType(ct CursorType)
//指定应为操作创建的光标类型。默认值是不可跟踪的，这意味着
//检索最后一批文档时，服务器将关闭光标。

SetHint(hint interface{})
//用于操作的索引。这应该是作为字符串的索引名或索引规范
//作为文件。如果提示参数是多键映射，驱动程序将返回一个错误。默认值为nil，
//这意味着不会发送任何提示。

SetLet(let interface{})
//指定查找表达式的参数。需要mongo版本大于等于5.0

SetLimit(i int64)
//要返回的最大文档数。默认值为0。

SetMax(max interface{})
//指定特定索引的排他性上界的文档。默认值为nil，这意味着没有最大值。

SetMaxAwaitTime(d time.Duration)
//服务器等待新文档满足可定制游标查询的最长时间。
//此选项仅适用于可定制的等待游标（有关更多信息，请参阅游标类型选项）和
//MongoDB版本>=3.2。对于其他游标类型或以前的服务器版本，此选项将被忽略。

SetMaxTime(d time.Duration)
//查询可以在服务器上运行的最长时间。默认值为nil，这意味着
//查询执行没有时间限制。

SetMin(min interface{})
//指定特定索引的包含下限的文档。默认值为0，这意味着
//没有最小值。

SetNoCursorTimeout(b bool)
//如果为true，则操作创建的光标在一段时间不活动后不会超时。默认值为false

SetOplogReplay(b bool)
//此选项仅供内部复制使用，不应设置。
//已弃用：此选项在MongoDB 4.4版中已弃用，如果使用，服务器将忽略它

SetProjection(projection interface{})
//描述操作返回的文档中将包含哪些字段的文档。默认值
//是nil，这意味着将包括所有字段。

SetReturnKey(b bool)
//如果为true，操作返回的文档将只包含与所用索引对应的字段。这个
//默认值为false。

SetShowRecordID(b bool)
//如果为true，则操作返回的文档中将包含带有记录标识符的$recordId字段。
//默认值为false。

SetSkip(i int64)
//将文档添加到结果之前要跳过的文档数。默认值为0。

SetSnapshot(b bool)
//如果为true，则由于中间的写入操作，光标不会多次返回文档。这个默认值为false。
//已弃用：此选项在MongoDB 3.6版中已弃用，并在MongoDB 4.0版中删除。

SetSort(sort interface{})
//一种文件，指定文件的归还顺序。如果排序参数是多键映射，则驱动程序将返回一个错误。

Update()

$rename

修改字段名

官方写法

db.students.updateOne(
   { _id: 1 },
   { $rename: { 'nickname': 'alias', 'cell': 'mobile' } }
)

golang写法

mongo.Api.UpdateMany(context.Background(), bson.M{}, bson.M{"$rename": bson.M{"explain": "group"}})

Aggregate()

project

修改字段

官方写法

{ $project: { <specification(s)> } }

golang写法

基础版本

filter := []bson.D{
		{{"$project", bson.D{
			{"_id", 0},
			{"group", "$_id"},
			{"apis", 1}}}},
	}
collection.Aggregate(context.Background(), filter)

修改对象里面的

filter := []bson.D{
		{{"$project", bson.D{
			{"_id", 0},
			{"group", "$_id"},
			{"apis", bson.D{{"group", 0}}}},//展示apis对象里面除group之外的，写1的话就是只展示group
	}
collection.Aggregate(context.Background(), filter)

修改数组对象里面的


filter := []bson.D{
		{{"$project", bson.D{
			{"_id", 0},
			{"group", "$_id"},
			{"apis", bson.D{{"group", 1}}}},//只展示apis数组的对象里面的group,不允许写0使用排除
	}
collection.Aggregate(context.Background(), filter)

group

分组

官方写法

{
  $group:
    {
      _id: <expression>, // 按表达式分组
      <field1>: { <accumulator1> : <expression1> },
      ...
    }
 }

golang写法

filter := []bson.D{
		{{"$group", bson.D{
			{"_id", bson.D{//这里的_id是分组的依据
				{"_id", "$_id"},
				{"devicename", "$devicename"},
				{"createTime", "$createTime"},
			}},
			{"key", bson.D{{"$push", "$key"}}},
		}}},
	}

arrayToObject

将数组转化成对象

前提

一个双元素数组的数组，其中第一个元素是字段名称，第二个元素是字段值：

[ [ "item", "abc123"], [ "qty", 25 ] ]

或

包含两个字段的文档数组，

该字段包含字段名称。k
该字段包含该字段的值。v

[ { "k": "item", "v": "abc123"}, { "k": "qty", "v": 25 } ]

官方写法

{ $arrayToObject: <expression> }

可以使用在$addFields，$project中

$project

{
   $project: {
      item: 1,
      dimensions: { $arrayToObject: "$dimensions" }
   }
}

$addFields

{ $addFields: { instock: { $arrayToObject: "$instock" } } }

golang写法

filter := []bson.D{
	{{"$project", bson.D{
		{"_id", 0},
		{"createTime", "$_id.createTime"},
		{"devicename", "$_id.devicename"},
		{"key", bson.D{{"$arrayToObject", "$key"}}},
	}}},
}

使用的方法

filter := []bson.D{
		{{"$unwind", bson.D{
			{"path", "$dataset1"},
		}}},
		{{"$project", bson.D{
			{"_id", 0},
			{"devicename", "$device.name"},
			{"key.k", "$dataset1.name"},
			{"key.v", "$dataset1.value"},
		}}},
		{{"$group", bson.D{
			{"_id", "$devicename"},
			{"key", bson.D{{"$push", "$key"}}},
		}}},
		{{"$project", bson.D{
			{"_id", 0},
			{"devicename", "$_id.devicename"},
			{"key", bson.D{{"$arrayToObject", "$key"}}},
		}}},
}
